CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE nexus_locks (
    task_id VARCHAR(255) PRIMARY KEY,
    node_id VARCHAR(100) NOT NULL,
    status VARCHAR(50) CHECK (status IN ('pending', 'running', 'completed', 'failed')),
    lock_ttl TIMESTAMP NOT NULL DEFAULT NOW() + INTERVAL '1 hour',
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE vector_knowledge (
    chunk_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    source_file VARCHAR(255),
    content TEXT NOT NULL,
    content_hash VARCHAR(64) UNIQUE NOT NULL, -- SHA-256 for Idempotency
    embedding vector(1536)
);

CREATE TABLE dead_letter_queue (
    error_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_name VARCHAR(255),
    payload JSONB,
    error_message TEXT,
    retry_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

import json
import hashlib
import logging
from supabase import create_client, Client
import requests # For local Ollama API

logging.basicConfig(filename='/var/log/nexus/ingestion.log', level=logging.INFO)

SUPABASE_URL = os.getenv("SUPABASE_URL")
SUPABASE_KEY = os.getenv("SUPABASE_KEY")
supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY)

def generate_hash(text: str) -> str:
    return hashlib.sha256(text.encode('utf-8')).hexdigest()

def get_ollama_embedding(text: str) -> list:
    try:
        response = requests.post('http://localhost:11434/api/embeddings', 
                                 json={"model": "mxbai-embed-large", "prompt": text})
        return response.json()['embedding']
    except Exception as e:
        logging.error(f"Ollama OOM/Timeout: {e}")
        raise

def process_chunk(chunk_text: str, source: str):
    chunk_hash = generate_hash(chunk_text)
    
    # Idempotency Check
    existing = supabase.table('vector_knowledge').select('content_hash').eq('content_hash', chunk_hash).execute()
    if existing.data:
        logging.info(f"Hash {chunk_hash} exists. Skipping deduplication.")
        return

    try:
        vector = get_ollama_embedding(chunk_text)
        supabase.table('vector_knowledge').insert({
            "source_file": source,
            "content": chunk_text,
            "content_hash": chunk_hash,
            "embedding": vector
        }).execute()
        logging.info(f"Ingested 1000-token chunk from {source}.")
    except Exception as e:
        # Send to DLQ
        supabase.table('dead_letter_queue').insert({
            "task_name": "vector_ingestion",
            "payload": {"source": source, "hash": chunk_hash},
            "error_message": str(e)
        }).execute()
